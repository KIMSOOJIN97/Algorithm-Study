#include <iostream>
#include <vector>
#include <map>
using namespace std;



int main(void) {

	freopen("inp.inp", "r", stdin);
	freopen("out.out", "w", stdout);

	long long n, m, k;
	cin >> n >> m >> k;
	vector<vector<long long >> list;
	vector<long long> box;
	for (int i = 0; i < n; i++) {
		int a;
		cin >> a;
		box.push_back(a);
	}
	list.push_back(box);

	int layer = 0;
	while (1) {
		if (list[layer].size() == 1)
			break;
		vector<long long> box;
		for (int i = 0; i < list[layer].size(); i += 2) {
			if (i >= list[layer].size()-1)
				break;
			box.push_back(list[layer][i] + list[layer][i + 1]);
		}
		if (list[layer].size() % 2 != 0) {
			box.push_back(list[layer][list[layer].size() - 1]);
		}
		list.push_back(box);
		layer++;
	}
	long long a, b, c;
	for (int i = n + 2; i <= n + m + k + 1;i++) {
		cin >> a >> b >> c;
		if (a == 1) {
			list[0][--b] = c;
			layer = 0;
			if (b % 2 != 0)
				b--;
			while (layer < list.size()-1) { 
				long long v = 0;
				if (b + 1 == list[layer].size())
					v = list[layer][b];
				else
					v = list[layer][b] + list[layer][b + 1];
				b /= 2;
				layer++;
				list[layer][b] = v;
				if (b % 2 != 0)
					b--;
			}
		}
		else {
			long long sum = 0; 
			b--;
			c--;
			if (b % 2 != 0&&b!=c) {
				sum += list[0][b];
				b++;
			}
			if (c % 2 == 0&&c!=b) {
				sum += list[0][c];
				c--;
			}
			layer = 0;
			while (b != c) {
				b /= 2;
				c /= 2;
				layer++;
			}
			sum += list[layer][b];
			cout << sum << endl;
		}
	}



	return 0;
}
